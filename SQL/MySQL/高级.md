
--- 七种 join 理论
-- 左连接
select <select_list>
from tableA A
left join TableB B
on A.Key = B.Key

-- 左外连接
select <select_list>
from TableA A
left join TableB B
on A.key=B.key
where B.key is null

-- 右连接
select <select_list>
from TableA A
right join TableB B
on A.key=B.key

-- 右外连接
select <select_list>
from TableA A
right join TableB B
on A.Key = B.Key
where A.Key is null

-- 内连接
select <select_list>
from TableA A
inner join TableB B
on A.Key = B.Key

-- 全连接
select <select_list>
from TableA A
full outer join TableB B
on A.Key = B.Key

-- 全外连接
select <select_list>
from Table A
full outer join TableB B
on A.Key = B.Key
where A.Key is null or B.Key is null
